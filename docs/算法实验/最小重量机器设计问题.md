```c++
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

struct HeapNode {
    int sum_weight = 0; //当前已选机器的重量和
    int sum_price = 0; //当前已选机器价值和
    vector<int> ret;//前level个商品选择的机器
    int val;//选择哪个供货商
    int level;//第几个部件
};

bool operator<(const HeapNode &a, const HeapNode &b) {
    return a.sum_weight > b.sum_weight;
}

int main() {
    //3 3 4 1 2 3 3 2 1 2 2 2 1 2 3 3 2 1 2 2 2
    int n, m, d;
    cin >> n >> m >> d;
    int weight[n + 1][m + 1];
    int price[n + 1][m + 1];
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= m; j++) {
            cin >> price[i][j];//价格
        }
    }
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= m; j++) {
            cin >> weight[i][j];//重量
        }
    }
    priority_queue<HeapNode, vector<HeapNode>> heap;//按照总重量排序的最小堆
    for(int i = 1; i <= m; i++) {
        HeapNode t;
        t.sum_weight = weight[1][i];
        t.sum_price = price[1][i];
        t.val = i;
        t.level = 1;
        t.ret.push_back(i);
        heap.emplace(t);
    }
    HeapNode now_t = heap.top();
    heap.pop();
    while(now_t.level < n) {//层数为n时，已经计算完
        for(int i = 1; i <= m; i++) {
            HeapNode t;
            t.sum_weight = now_t.sum_weight + weight[now_t.level + 1][i];
            t.sum_price = now_t.sum_price + price[now_t.level + 1][i];
            t.val = i;
            t.level = now_t.level + 1;
            t.ret = now_t.ret;
            t.ret.push_back(i);
            if(t.sum_price <= d) {
                heap.push(t);
            }
        }
        now_t = heap.top();
        heap.pop();
    }
    cout << now_t.sum_weight << endl;
    for(int i : now_t.ret) {
        cout << i << " ";
    }
    cout << endl;
    return 0;
}
```
